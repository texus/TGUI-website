208 template <typename Func, typename... BoundArgs>
209 unsigned int SignalManager::connect(String widgetName, String signalName, Func&& handler, const BoundArgs&... args)
210 {
211 const unsigned int id = generateUniqueId();
212
213 if constexpr (std::is_convertible_v<Func, std::function<void(const BoundArgs&...)>>
214 && std::is_invocable_v<decltype(&handler), BoundArgs...>
215 && !std::is_function_v<Func>)
216 {
217 // Reference to function
218 m_signals[id] = {widgetName, signalName, makeSignal([=, f=std::function<void(const BoundArgs&...)>(handler)]{ std::invoke(f, args...); })};
219 }
220 else if constexpr (std::is_convertible_v<Func, std::function<void(const BoundArgs&...)>>)
221 {
222 // Function pointer
223 m_signals[id] = {widgetName, signalName, makeSignal([=]{ std::invoke(handler, args...); })};
224 }
225 else if constexpr (std::is_convertible_v<Func, std::function<void(const BoundArgs&..., const std::shared_ptr<Widget>&, const String&)>>
226 && std::is_invocable_v<decltype(&handler), BoundArgs..., const std::shared_ptr<Widget>&, const String&>
227 && !std::is_function_v<Func>)
228 {
229 // Reference to function with caller arguments
230 m_signals[id] = {widgetName, signalName, makeSignalEx([=, f=std::function<void(const BoundArgs&..., const std::shared_ptr<Widget>& w, const String& s)>(handler)](const std::shared_ptr<Widget>& w, const String& s){ std::invoke(f, args..., w, s); })};
231 }
232 else
233 {
234 // Function pointer with caller arguments
235 m_signals[id] = {widgetName, signalName, makeSignalEx([=](const std::shared_ptr<Widget>& w, const String& s){ std::invoke(handler, args..., w, s); })};
236 }
237
238 connect(id);
239 return id;
240 }
241#else
